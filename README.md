```yaml
menu:
  animation:
    enabled: true               # Вкл/выкл анимацию при открытии меню. Если не нужна — ставьте false.
    type: SPIRAL                # Вид анимации: fill_glass — просто заполняет слоты, wave — идёт волной, random_fill — раскидывает рандомно, spiral — закручивается по спирали.
    direction: FORWARD          # Куда "двигается" анимация: forward — слева направо и сверху вниз, backward — наоборот, random — как повезёт.
    material: "LIGHT_BLUE_STAINED_GLASS_PANE"  # Какой блок использовать для временного заполнения слотов.
    speed: 3                    # Задержка между шагами (в тиках). 20 тиков = 1 сек. Меньше — быстрее.
    sound: BLOCK_NOTE_BLOCK_BANJO  # Какой звук проигрывать при заполнении слота.
    sound_volume: 1.0           # Громкость звука.
    sound_pitch: 1.2            # Высота звука.
    batch_size: 3               # Сколько слотов заполняется за один тик. Чем больше — тем быстрее проходит анимация.
    autoUpdate: false           # Автообновление меню каждые 60 тиков. Если не нужно — оставляйте false.

```
